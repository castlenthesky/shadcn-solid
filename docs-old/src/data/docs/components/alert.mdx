## Usage

```tsx
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";
```

```tsx
<Alert>
  <TbTerminal class="h-4 w-4" />
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can not add components and dependencies to your app using the cli, yet.
  </AlertDescription>
</Alert>
```
