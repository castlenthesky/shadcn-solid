## Usage

```tsx
import { toast } from "somoto"
```

```tsx
toast("Event has been created.")
```

```tsx title="src/main.tsx" {1,11}
import { Toaster } from "@/components/ui/sonner"
 
const App = () => {
  return (
    <Router
      root={props => (
          <Suspense>
            <ColorModeScript />
            <ColorModeProvider>
              {props.children}
              <Toaster />
            </ColorModeProvider>
          </Suspense>
      )}
    >
      <FileRoutes />
    </Router>
  );
};
 
export default App;
```
